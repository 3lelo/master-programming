مرحبًا بالعالم باستخدام كوتلن

برنامج "مرحبًا بالعالم" هو أبسط طريقة للبدء بالبرمجة في كوتلن.

// مرحبًا بالعالم في كوتلن
fun main() {
    println("Hello, World!")
}

هذا البرنامج يُعرّف دالة main، وهي نقطة البداية للتطبيق. وتُستخدم دالة println لطباعة النصوص على وحدة التحكم.

أنواع البيانات في كوتلن

تُقدم كوتلن أنواع بيانات متعددة لتخزين قيم مختلفة:

  • Int: يخزن الأعداد الصحيحة (مثل: val x: Int = 10)
  • Double: يخزن الأعداد العشرية (مثل: val y: Double = 3.14)
  • Char: يخزن حرفًا واحدًا (مثل: val c: Char = 'A')
  • Boolean: يخزن القيم المنطقية (صح أو خطأ) (مثل: val isKotlinFun: Boolean = true)
  • String: يخزن النصوص (مثل: val message: String = "Hello")

إليك مثالًا لاستخدام هذه الأنواع في كوتلن:

                        
fun main() {
    val age: Int = 25
    val pi: Double = 3.14159
    val grade: Char = 'A'
    val isActive: Boolean = true
    val name: String = "John"

    println("العمر: $age")
    println("قيمة π: $pi")
    println("التقدير: $grade")
    println("هل نشط؟: $isActive")
    println("الاسم: $name")
}

العوامل في كوتلن

توفر كوتلن مجموعة غنية من العوامل لإجراء عمليات مختلفة على المتغيرات والقيم، مع ميزات فريدة مقارنة بجافا.

العوامل الحسابية

// العمليات الحسابية الأساسية
val a = 10
val b = 3

val sum = a + b       // الجمع: 13
val difference = a - b// الطرح: 7
val product = a * b   // الضرب: 30
val division = a / b  // القسمة العددية: 3
val modulus = a % b   // الباقي: 1

// عامل القسمة الآمنة
val safeDivision = a.div(b)  // قسمة آمنة في كوتلن

عوامل المقارنة

// عوامل المقارنة
val x = 5
val y = 10

val isEqual = x == y       // المساواة الهيكلية
val isExactEqual = x === y // المساواة المرجعية
val isNotEqual = x != y    // غير متساوي
val isGreater = x > y      // أكبر من
val isLess = x < y         // أصغر من
val isGreaterOrEqual = x >= y
val isLessOrEqual = x <= y

العوامل المنطقية

// العوامل المنطقية
val isTrue = true
val isFalse = false

val andResult = isTrue && isFalse   // و منطقي
val orResult = isTrue || isFalse    // أو منطقي
val notResult = !isTrue             // نفي منطقي

عوامل النطاق والانتماء

// عوامل فريدة في كوتلن
val range = 1..10         // إنشاء نطاق من 1 إلى 10
val inRange = 5 in range  // تحقق إذا كانت 5 ضمن النطاق: true

val list = listOf(1, 2, 3, 4, 5)
val containsCheck = 3 in list   // تحقق الانتماء: true
val notContainsCheck = 6 !in list // عكس الانتماء: true

// زيادة وتنقيص
var counter = 0
counter++   // زيادة بعدية
++counter   // زيادة قبلية
counter--   // تنقيص بعدي
--counter   // تنقيص قبلي

العوامل الثنائية

// العمليات الثنائية
val num1 = 5  // ثنائي: 0101
val num2 = 3  // ثنائي: 0011

val bitwiseAnd = num1 and num2   // و ثنائي: 1
val bitwiseOr = num1 or num2     // أو ثنائي: 7
val bitwiseXor = num1 xor num2   // أو حصر ثنائي: 6
val bitwiseInvert = num1.inv()   // نفي ثنائي

عامل الانتشار وعامل Elvis

// عامل الانتشار
fun printNumbers(vararg numbers: Int) {
    numbers.forEach { println(it) }
}
val nums = intArrayOf(1, 2, 3)
printNumbers(*nums)  // عامل الانتشار يفكك المصفوفة

// عامل Elvis (التحقق من القيم الفارغة)
val nullableStr: String? = null
val definedStr = nullableStr ?: "قيمة افتراضية"

نصائح عن عوامل كوتلن:
  • كوتلن توفر عوامل أكثر أمانًا من حيث النوع ووعيًا بالقيم الفارغة
  • استخدم === للمساواة المرجعية، و == للمساواة الهيكلية
  • استفد من العوامل الآمنة للتعامل مع الأنواع القابلة لأن تكون فارغة

ميزات النصوص في كوتلن:
  • تدعم القوالب النصية للتضمين السهل
  • توفر طرقًا مدمجة شاملة لتلاعب النصوص
  • غير قابلة للتغيير من أجل أمان الخيوط

الرياضيات في كوتلن

توفر كوتلن عمليات رياضية قوية وتدعم الحسابات الرياضية القياسية.

fun main() {
    // العمليات الحسابية الأساسية
    val a = 10
    val b = 3

    // الجمع
    println("الجمع: ${a + b}")  // 13

    // الطرح
    println("الطرح: ${a - b}")  // 7

    // الضرب
    println("الضرب: ${a * b}")  // 30

    // القسمة
    println("القسمة: ${a / b}")  // 3 (قسمة عددية)
    println("قسمة دقيقة: ${a.toDouble() / b}")  // 3.333...

    // الباقي (المدلول)
    println("الباقي: ${a % b}")  // 1

    // دوال رياضية
    println("القيمة المطلقة: ${Math.abs(-5)}")  // 5
    println("القيمة العظمى: ${Math.max(a, b)}")  // 10
    println("القيمة الصغرى: ${Math.min(a, b)}")  // 3
    println("الأس: ${Math.pow(2.0, 3.0)}")  // 8.0
    println("الجذر التربيعي: ${Math.sqrt(16.0)}")  // 4.0
}

الميزات الرئيسية للرياضيات في كوتلن:
  • تدعم العمليات الحسابية القياسية (+, -, *, /, %)
  • توفر فئة Math مع طرق مساعدة
  • تدعم تحويل الأنواع للحسابات الدقيقة
  • عمليات رياضية آمنة من حيث النوع

القيم البوليانية في كوتلن

تمثل القيم البوليانية في كوتلن قيم الحقيقة (true/false) مع عمليات منطقية شاملة.

fun main() {
    // تعريف القيم البوليانية
    val isTrue = true
    val isFalse = false

    // العمليات المنطقية
    println("عملية AND: ${isTrue && isFalse}")  // false
    println("عملية OR: ${isTrue || isFalse}")   // true
    println("عملية NOT: ${!isTrue}")            // false

    // العمليات المقارنة
    val x = 5
    val y = 10
    println("يساوي: ${x == y}")        // false
    println("لا يساوي: ${x != y}")    // true
    println("أكبر من: ${x > y}")      // false
    println("أصغر من: ${x < y}")      // true
    println("أكبر أو يساوي: ${x >= y}") // false
    println("أصغر أو يساوي: ${x <= y}") // true
}

عمليات القيم البوليانية في كوتلن تشمل:
  • العوامل المنطقية: && (AND)، || (OR)، ! (NOT)
  • عوامل المقارنة لمختلف الأنواع
  • منطق بولياني آمن وصارم

تعليمات If ... Else في كوتلن

توفر تعليمات if-else في كوتلن مرونة وقوة في الاستخدام.

fun main() {
    val age = 20

    // تعليمة if بسيطة
    if (age >= 18) {
        println("أنت بالغ")
    }

    // تعليمة if-else
    if (age >= 18) {
        println("يمكنك التصويت")
    } else {
        println("لا يمكنك التصويت")
    }

    // سلم if-else-if
    val score = 85
    when {
        score >= 90 -> println("ممتاز")
        score >= 80 -> println("جيد")
        score >= 70 -> println("مرضٍ")
        else -> println("يحتاج إلى تحسين")
    }

    // تعليمة if-else مختصرة (مثل ternary)
    val result = if (age >= 18) "بالغ" else "قاصر"
    println(result)
}

ميزات تعليمات if-else:
  • صياغة if-else التقليدية
  • يمكن استخدامها كتعابير
  • دعم للعديد من الحالات الشرطية
  • تعيينات شرطية مضمنة

تعليمة When (بديل Switch) في كوتلن

تعليمة when في كوتلن هي بديل قوي لتعليمة switch.

fun main() {
    // تعبير when الأساسي
    val day = 3
    when (day) {
        1 -> println("الإثنين")
        2 -> println("الثلاثاء")
        3 -> println("الأربعاء")
        4 -> println("الخميس")
        5 -> println("الجمعة")
        6 -> println("السبت")
        7 -> println("الأحد")
        else -> println("يوم غير صالح")
    }

    // قيم متعددة في فرع واحد
    val x = 1
    when (x) {
        0, 1 -> println("صفر أو واحد")
        else -> println("شيء آخر")
    }

    // التحقق من النطاق والنوع
    val value: Any = "مرحباً"
    when (value) {
        in 1..10 -> println("رقم بين 1 و 10")
        is String -> println("إنها سلسلة نصية")
        !is Int -> println("ليس عدد صحيح")
    }
}

قدرات تعليمة when:
  • أكثر مرونة من switch التقليدية
  • يمكنها مطابقة قيم متعددة
  • تدعم التحقق من النطاق والنوع
  • يمكن استخدامها كتعبير

الحلقات في كوتلن

توفر كوتلن أنواعًا متعددة من الحلقات لتناسب مختلف السيناريوهات.

fun main() {
    // حلقة for مع نطاق
    for (i in 1..5) {
        println(i)  // يطبع 1, 2, 3, 4, 5
    }

    // حلقة for مع مصفوفة
    val fruits = arrayOf("Apple", "Banana", "Cherry")
    for (fruit in fruits) {
        println(fruit)
    }

    // حلقة while
    var count = 0
    while (count < 5) {
        println(count)
        count++
    }

    // حلقة do-while
    var x = 0
    do {
        println(x)
        x++
    } while (x < 3)

    // التكرار مع الفهرس
    for ((index, value) in fruits.withIndex()) {
        println("Index: $index, Value: $value")
    }
}

أنواع الحلقات في كوتلن:
  • حلقات for مع النطاقات والمجموعات
  • حلقات while وdo-while
  • تكرار سهل مع الفهرس والقيمة
  • أكثر اختصارًا وسهولة في القراءة

التعليمات Break و Continue في كوتلن

تحكم في تنفيذ الحلقات باستخدام التعليمات break و continue.

fun main() {
    // تعليمة break داخل حلقة
    for (i in 1..10) {
        if (i == 5) {
            break  // يخرج من الحلقة تمامًا
        }
        println(i)
    }

    // تعليمة continue داخل حلقة
    for (i in 1..5) {
        if (i == 3) {
            continue  // يتخطى بقية التكرار الحالي
        }
        println(i)
    }

    // تعليمة break و continue مع التسمية
    outerLoop@ for (i in 1..3) {
        for (j in 1..3) {
            if (i == 2 && j == 2) {
                break@outerLoop  // يخرج من الحلقة الخارجية
            }
            println("i = $i, j = $j")
        }
    }
}

ميزات التعليمات break و continue:
  • break تخرج من الحلقة بالكامل
  • continue تتخطى التكرار الحالي
  • دعم للتعليمات المسماة
  • مفيدة في سيناريوهات الحلقات المتداخلة

الدوال في كوتلن

تُستخدم الدوال (Functions) في كوتلن لتنظيم الكود وجعله أكثر قابلية لإعادة الاستخدام. يتم تعريف الدالة مرة واحدة ثم استدعاؤها عدة مرات عند الحاجة.

إنشاء دالة

// إنشاء دالة تقوم بطباعة رسالة
fun sayHello() {
    println("Hello, Kotlin!")
}

// استدعاء الدالة
fun main() {
    sayHello()
}

دالة تُرجع قيمة

// دالة تقوم بجمع رقمين وتُرجع النتيجة
fun add(num1: Int, num2: Int): Int {
    return num1 + num2
}

// استدعاء الدالة
fun main() {
    val result = add(5, 3)
    println("Result: $result")
}

دالة لا تُرجع قيمة (Unit)

// دالة تقوم بطباعة رسالة بدون إرجاع قيمة
fun printMessage() {
    println("This is a Unit function!")
}

// استدعاء الدالة
fun main() {
    printMessage()
}

تمرير المعاملات إلى الدالة

// دالة تقوم بحساب مساحة المستطيل
fun calculateArea(length: Int, width: Int): Int {
    return length * width
}

// استدعاء الدالة
fun main() {
    val area = calculateArea(5, 10)
    println("Area: $area")
}

الدوال المُثقلة (Function Overloading)

// تعريف دالتين بنفس الاسم ولكن بمعاملات مختلفة
fun multiply(a: Int, b: Int): Int {
    return a * b
}

fun multiply(a: Double, b: Double): Double {
    return a * b
}

// استدعاء الدوال
fun main() {
    println("Multiplication (Int): ${multiply(4, 5)}")
    println("Multiplication (Double): ${multiply(4.5, 3.2)}")
}

الدوال العُليا (Higher-Order Functions)

// دالة تأخذ دالة أخرى كمعامل
fun operateOnNumbers(a: Int, b: Int, operation: (Int, Int) -> Int): Int {
    return operation(a, b)
}

// استدعاء الدالة مع تمرير دالة مجهولة
fun main() {
    val sum = operateOnNumbers(5, 3) { x, y -> x + y }
    println("Sum: $sum")
}

الدوال المُدرجة (Lambda Functions)

// دالة مجهولة (Lambda)
val square: (Int) -> Int = { number -> number * number }

// استدعاء الدالة
fun main() {
    println("Square of 4: ${square(4)}")
}

نصيحة احترافية: توفر كوتلن ميزات قوية مثل الدوال العُليا (Higher-Order Functions) والدوال المجهولة (Lambdas) لجعل الكود أكثر مرونة وكفاءة.

الفئات والكائنات (Classes and Objects)

تعتبر الفئات (Classes) والكائنات (Objects) الأساس في البرمجة كائنية التوجه. الفئة هي مخطط (Blueprint) لإنشاء الكائنات، بينما الكائن هو نسخة من هذه الفئة. يمكن أن تحتوي الفئات على خصائص (Properties) ودوال (Functions) التي تحدد سلوك الكائنات.

في كوتلن، يمكن تعريف الفئات باستخدام الكلمة المفتاحية class، ويمكننا إنشاء كائنات منها باستخدام المُنشئ الافتراضي.

// تعريف فئة تحتوي على خاصية ودالة
class Person(val name: String) {
    fun greet() {
        println("Hello, my name is $name")
    }
}
// إنشاء كائن من الفئة واستدعاء الدالة
fun main() {
    val person = Person("Ahmed")
    person.greet()
}

يمكننا أيضًا تعريف خصائص قابلة للتغيير باستخدام var، بينما يتم تعريف الخصائص غير القابلة للتغيير باستخدام val. كما يمكن تعيين قيم افتراضية للخصائص.

// تعريف فئة بخصائص قابلة للتغيير وثابتة
class Car(var model: String, val year: Int) {
    fun displayInfo() {
        println("Car Model: $model, Year: $year")
    }
}
// إنشاء كائن وتعديل خصائصه
fun main() {
    val car = Car("Toyota", 2023)
    car.displayInfo()
    car.model = "Honda" // يمكن تغييره لأنه var
    car.displayInfo()
}

كما يمكننا إنشاء كائنات بدون تعريف فئة مسبقًا باستخدام object، وهو ما يعرف باسم الكائنات المجهولة (Anonymous Objects).

// كائن مجهول يحتوي على دالة
fun main() {
    val obj = object {
        fun sayHello() {
            println("Hello from an anonymous object!")
        }
    }
    obj.sayHello()
}

المُنشئات الأولية والثانوية (Primary and Secondary Constructors)

تستخدم كوتلن نوعين من المُنشئات لإنشاء الكائنات: المُنشئ الأولي (Primary Constructor) والمُنشئ الثانوي (Secondary Constructor). يمكن استخدام المُنشئات لتهيئة الخصائص عند إنشاء الكائن.

المُنشئ الأولي يُعرَّف مباشرةً في رأس الفئة بعد اسمها، ويمكن استخدامه لتهيئة الخصائص دون الحاجة إلى كتلة تنفيذية.

// تعريف فئة بمُنشئ أولي
class Person(val name: String, val age: Int) {
    fun introduce() {
        println("Hello, my name is $name and I am $age years old.")
    }
}
// إنشاء كائن باستخدام المُنشئ الأولي
fun main() {
    val person = Person("Ali", 25)
    person.introduce()
}

أما المُنشئ الثانوي فيتم تعريفه داخل جسم الفئة باستخدام الكلمة المفتاحية constructor. يُستخدم المُنشئ الثانوي عند الحاجة إلى تنفيذ منطق إضافي أثناء التهيئة.

// تعريف فئة بمُنشئ أولي وثانوي
class Car() {
    val brand: String
    val model: String
    
    // المُنشئ الأولي
    constructor(brand: String, model: String) {
        this.brand = brand
        this.model = model
    }
    
    // المُنشئ الثانوي
    constructor(brand: String) : this(brand, "Unknown") {
        println("Secondary constructor called")
    }
}
// إنشاء كائنات باستخدام المُنشئات
fun main() {
    val car1 = Car("Toyota", "Corolla")
    val car2 = Car("Honda")
}

يمكن أيضًا استخدام القيم الافتراضية في المُنشئ الأولي لتقليل الحاجة إلى مُنشئات ثانوية.

// استخدام القيم الافتراضية بدلاً من المُنشئ الثانوي
class Laptop(val brand: String, val model: String = "Unknown") {
    fun showInfo() {
        println("Laptop Brand: $brand, Model: $model")
    }
}
// إنشاء كائنات باستخدام القيم الافتراضية
fun main() {
    val laptop1 = Laptop("Dell", "XPS 15")
    val laptop2 = Laptop("HP")
    laptop1.showInfo()
    laptop2.showInfo()
}

الوراثة (Inheritance)

الوراثة (Inheritance) هي ميزة تسمح للفئات باكتساب خصائص ودوال من فئات أخرى، مما يساعد في إعادة استخدام الكود.

// تعريف فئة رئيسية وفئة فرعية
open class Animal(val name: String) {
    open fun makeSound() {
        println("$name makes a sound")
    }
}
class Dog(name: String) : Animal(name) {
    override fun makeSound() {
        println("$name barks")
    }
}
// استدعاء الوراثة
fun main() {
    val dog = Dog("Buddy")
    dog.makeSound()
}

الوراثة (Inheritance) - تجاوز الدوال (Overriding)

يمكن للفئات الفرعية (Subclass) تجاوز الدوال في الفئة الرئيسية (Superclass) باستخدام الكلمة المفتاحية override.

// تجاوز دالة من الفئة الأساسية
open class Animal {
    open fun makeSound() {
        println("Animal makes a sound")
    }
}
class Cat : Animal() {
    override fun makeSound() {
        println("Meow!")
    }
}
// استدعاء الدالة
fun main() {
    val cat = Cat()
    cat.makeSound()
}

الواجهات (Interfaces)

الواجهات (Interfaces) في كوتلن تُستخدم لتعريف سلوكيات مشتركة يمكن أن تتبناها عدة فئات دون الحاجة إلى الوراثة المباشرة. تحتوي الواجهة على دوال بدون تنفيذ افتراضي، لكن يمكن أن تحتوي على تنفيذ افتراضي أيضًا.

يمكن لأي فئة تطبيق واجهة باستخدام الكلمة المفتاحية interface، كما يمكن للفئة أن تطبق أكثر من واجهة واحدة.

// تعريف واجهة تحتوي على دالة مجردة
interface Drivable {
    fun drive()
}
// تطبيق الواجهة في فئة
class Car : Drivable {
    override fun drive() {
        println("The car is driving")
    }
}
// استدعاء الواجهة
fun main() {
    val myCar = Car()
    myCar.drive()
}

يمكن أن تحتوي الواجهات أيضًا على دوال ذات تنفيذ افتراضي، مما يسمح بإعادة استخدام الكود عبر الفئات المختلفة.

// واجهة تحتوي على دالة افتراضية
interface Animal {
    fun makeSound() {
        println("Animal makes a sound")
    }
}
// تطبيق الواجهة في فئة
class Dog : Animal {
    override fun makeSound() {
        println("Woof! Woof!")
    }
}
// استدعاء الكائن
fun main() {
    val dog = Dog()
    dog.makeSound()
}

يمكن للفئة أن تطبق أكثر من واجهة، مما يوفر مرونة أكبر في التصميم.

// تعريف واجهتين
interface Flyable {
    fun fly()
}
interface Swimmable {
    fun swim()
}
// فئة تطبق أكثر من واجهة
class Duck : Flyable, Swimmable {
    override fun fly() {
        println("Duck is flying")
    }
    override fun swim() {
        println("Duck is swimming")
    }
}
// استدعاء الدوال
fun main() {
    val duck = Duck()
    duck.fly()
    duck.swim()
}

الفئات المجردة (Abstract Classes)

الفئات المجردة (Abstract Classes) تُستخدم عندما نحتاج إلى فئة تحتوي على دوال غير معرفة يجب على الفئات الفرعية تنفيذها. لا يمكن إنشاء كائنات مباشرة من الفئات المجردة، بل يجب أن يتم توسيعها (Extend) من خلال فئات أخرى.

تُعرّف الفئة المجردة بالكلمة المفتاحية abstract، ويمكن أن تحتوي على دوال مجردة (بدون تنفيذ) أو دوال عادية مع تنفيذ افتراضي.

// تعريف فئة مجردة ودالة مجردة
abstract class Vehicle {
    abstract fun startEngine()
    fun stopEngine() {
        println("Engine stopped")
    }
}
// فئة فرعية تطبق الفئة المجردة
class Bike : Vehicle() {
    override fun startEngine() {
        println("Bike engine started")
    }
}
// استدعاء الفئة المجردة
fun main() {
    val myBike = Bike()
    myBike.startEngine()
    myBike.stopEngine()
}

يمكن أيضًا أن تحتوي الفئات المجردة على خصائص مجردة يجب تهيئتها في الفئات الفرعية.

// تعريف فئة مجردة بخصائص مجردة
abstract class Animal {
    abstract val name: String
    abstract fun makeSound()
}
// فئة فرعية تطبق الفئة المجردة
class Dog : Animal() {
    override val name = "Dog"
    override fun makeSound() {
        println("Woof! Woof!")
    }
}
// استدعاء الكائن
fun main() {
    val dog = Dog()
    println("Animal: ${dog.name}")
    dog.makeSound()
}

مشروع 1: نظام إدارة المكتبة (Library Management System)


// تعريف كلاس الكتاب مع خصائص متعددة
class Book(
    val id: Int,
    val title: String,
    val author: String,
    val yearPublished: Int,
    var isAvailable: Boolean = true
) {
    // طريقة لعرض معلومات الكتاب
    fun displayInfo() {
        println("كتاب: $title")
        println("المؤلف: $author")
        println("سنة النشر: $yearPublished")
        println("متوفر: ${if (isAvailable) "نعم" else "لا"}")
    }
    
    // طريقة لاستعارة الكتاب
    fun borrowBook(): Boolean {
        return if (isAvailable) {
            isAvailable = false
            println("تم استعارة الكتاب: $title")
            true
        } else {
            println("الكتاب غير متوفر للاستعارة")
            false
        }
    }
    
    // طريقة لإرجاع الكتاب
    fun returnBook() {
        if (!isAvailable) {
            isAvailable = true
            println("تم إرجاع الكتاب: $title")
        } else {
            println("هذا الكتاب موجود بالفعل في المكتبة!")
        }
    }
}

// كلاس المكتبة يدير مجموعة من الكتب
class Library {
    private val books = mutableListOf<Book>()
    
    // إضافة كتاب جديد
    fun addBook(book: Book) {
        books.add(book)
        println("تمت إضافة كتاب: ${book.title}")
    }
    
    // البحث عن كتاب بواسطة العنوان
    fun findBookByTitle(title: String): Book? {
        return books.find { it.title.contains(title, ignoreCase = true) }
    }
    
    // عرض جميع الكتب
    fun displayAllBooks() {
        if (books.isEmpty()) {
            println("لا توجد كتب في المكتبة")
            return
        }
        
        println("=== قائمة الكتب ===")
        books.forEach { book ->
            book.displayInfo()
            println("------------------")
        }
    }
    
    // عرض الكتب المتاحة فقط
    fun displayAvailableBooks() {
        val availableBooks = books.filter { it.isAvailable }
        
        if (availableBooks.isEmpty()) {
            println("لا توجد كتب متاحة حاليًا")
            return
        }
        
        println("=== الكتب المتاحة ===")
        availableBooks.forEach { book ->
            book.displayInfo()
            println("------------------")
        }
    }
}

// اختبار النظام
fun main() {
    val library = Library()
    
    // إنشاء بعض الكتب
    val book1 = Book(1, "البؤساء", "فيكتور هوغو", 1862)
    val book2 = Book(2, "الحرب والسلام", "ليو تولستوي", 1869)
    val book3 = Book(3, "جريمة وعقاب", "فيودور دوستويفسكي", 1866)
    
    // إضافة الكتب إلى المكتبة
    library.addBook(book1)
    library.addBook(book2)
    library.addBook(book3)

    library.displayAllBooks()

    val foundBook = library.findBookByTitle("البؤساء")

    // عرض الكتب المتاحة بعد الاستعارة
    library.displayAvailableBooks()
    
    // إرجاع الكتاب
    foundBook?.returnBook()
    
    // عرض الكتب المتاحة بعد الإرجاع
    library.displayAvailableBooks()
}

شرح المشروع:

هذا المشروع يمثل نظام إدارة مكتبة بسيط باستخدام الكلاسات والكائنات في كوتلن. يتكون من كلاسين رئيسيين:

  • Book: يمثل كتابًا فرديًا مع خصائص مثل العنوان والمؤلف وسنة النشر، بالإضافة إلى طرق لاستعارة وإرجاع الكتاب.
  • Library: يدير مجموعة من الكتب ويوفر وظائف مثل إضافة كتاب، والبحث عن كتاب، وعرض جميع الكتب أو الكتب المتاحة فقط.

المشروع يوضح المفاهيم الأساسية للكلاسات والكائنات: تعريف الخصائص، إنشاء الطرق، إنشاء كائنات، واستخدام الكائنات للتفاعل مع بعضها البعض.

مشروع 2: نظام إدارة الموظفين (Employee Management System)


// تعريف كلاس الموظف
class Employee(
    val id: Int,
    val name: String,
    val position: String,
    private var salary: Double,
    val hireDate: String
) {
    // خاصية محسوبة لتحديد ما إذا كان الموظف جديدًا
    val isNewEmployee: Boolean
        get() {
            // نفترض أنه تم توظيفه في عام 2023 أو بعده
            return hireDate.endsWith("2023") || hireDate.endsWith("2024")
        }
    
    // دالة معدلة (mutator) لتغيير الراتب
    fun updateSalary(newSalary: Double) {
        if (newSalary > salary) {
            println("تمت زيادة راتب $name من $salary إلى $newSalary")
            salary = newSalary
        } else {
            println("الراتب الجديد يجب أن يكون أكبر من الراتب الحالي")
        }
    }
    
    // زيادة الراتب بنسبة مئوية
    fun giveRaise(percentage: Double) {
        val raiseAmount = salary * (percentage / 100)
        val newSalary = salary + raiseAmount
        println("تمت زيادة راتب $name بنسبة $percentage% (${raiseAmount})")
        salary = newSalary
    }
    
    // عرض معلومات الموظف
    fun displayInfo() {
        println("معرف: $id")
        println("الاسم: $name")
        println("المنصب: $position")
        println("الراتب: $salary")
        println("تاريخ التوظيف: $hireDate")
        println("موظف جديد: ${if (isNewEmployee) "نعم" else "لا"}")
    }
}

// كلاس لإدارة الموظفين
class EmployeeManager {
    private val employees = mutableListOf<Employee>()
    
    // إضافة موظف جديد
    fun addEmployee(employee: Employee) {
        employees.add(employee)
        println("تمت إضافة الموظف: ${employee.name}")
    }
    
    // إزالة موظف
    fun removeEmployee(id: Int): Boolean {
        val employee = findEmployeeById(id)
        return if (employee != null) {
            employees.remove(employee)
            println("تمت إزالة الموظف: ${employee.name}")
            true
        } else {
            println("لم يتم العثور على موظف بهذا المعرف")
            false
        }
    }
    
    // البحث عن موظف بالمعرف
    fun findEmployeeById(id: Int): Employee? {
        return employees.find { it.id == id }
    }
    
    // عرض جميع الموظفين
    fun displayAllEmployees() {
        if (employees.isEmpty()) {
            println("لا يوجد موظفين")
            return
        }
        
        println("=== قائمة الموظفين ===")
        employees.forEach { employee ->
            employee.displayInfo()
            println("------------------")
        }
    }
    
    // حساب متوسط الرواتب
    fun calculateAverageSalary(): Double {
        if (employees.isEmpty()) return 0.0
        val totalSalary = employees.sumOf { it.javaClass.getDeclaredField("salary").apply { isAccessible = true }.getDouble(it) }
        return totalSalary / employees.size
    }
    
    // إعطاء زيادة لجميع الموظفين
    fun giveRaiseToAll(percentage: Double) {
        employees.forEach { it.giveRaise(percentage) }
        println("تم إعطاء زيادة بنسبة $percentage% لجميع الموظفين")
    }
    
    // عرض الموظفين الجدد فقط
    fun displayNewEmployees() {
        val newEmployees = employees.filter { it.isNewEmployee }
        
        if (newEmployees.isEmpty()) {
            println("لا يوجد موظفين جدد")
            return
        }
        
        println("=== الموظفين الجدد ===")
        newEmployees.forEach { employee ->
            employee.displayInfo()
            println("------------------")
        }
    }
}

هذا المشروع يمثل نظام إدارة الموظفين يستخدم مفاهيم متقدمة في الكلاسات والكائنات:

مشروع 3: محاكاة نظام بنكي (Banking System Simulation)


// تعريف كلاس الحساب البنكي
class BankAccount(
    val accountNumber: String,
    val ownerName: String,
    private var balance: Double = 0.0
) {
    // سجل المعاملات
    private val transactions = mutableListOf<String>()
    
    init {
        // تسجيل إنشاء الحساب
        addTransaction("تم فتح الحساب برصيد أولي: $balance")
    }
    
    // إضافة معاملة إلى السجل
    private fun addTransaction(description: String) {
        val timestamp = java.time.LocalDateTime.now().toString()
        transactions.add("[$timestamp] $description")
    }
    
    // إيداع أموال
    fun deposit(amount: Double): Boolean {
        if (amount <= 0) {
            println("يجب أن يكون مبلغ الإيداع أكبر من الصفر")
            return false
        }
        
        balance += amount
        addTransaction("إيداع: $amount، الرصيد الجديد: $balance")
        println("تم إيداع $amount. الرصيد الحالي: $balance")
        return true
    }
    
    // سحب أموال
    fun withdraw(amount: Double): Boolean {
        if (amount <= 0) {
            println("يجب أن يكون مبلغ السحب أكبر من الصفر")
            return false
        }
        
        if (amount > balance) {
            println("رصيد غير كافٍ. الرصيد الحالي: $balance")
            return false
        }
        
        balance -= amount
        addTransaction("سحب: $amount، الرصيد الجديد: $balance")
        println("تم سحب $amount. الرصيد الحالي: $balance")
        return true
    }
    
    // تحويل أموال إلى حساب آخر
    fun transfer(targetAccount: BankAccount, amount: Double): Boolean {
        if (amount <= 0) {
            println("يجب أن يكون مبلغ التحويل أكبر من الصفر")
            return false
        }
        
        if (amount > balance) {
            println("رصيد غير كافٍ للتحويل. الرصيد الحالي: $balance")
            return false
        }
        
        balance -= amount
        targetAccount.deposit(amount)
        
        addTransaction("تحويل: $amount إلى حساب ${targetAccount.accountNumber} (${targetAccount.ownerName})، الرصيد الجديد: $balance")
        println("تم تحويل $amount إلى حساب ${targetAccount.ownerName}. الرصيد الحالي: $balance")
        return true
    }
    
    // الحصول على الرصيد الحالي
    fun getBalance(): Double {
        return balance
    }
    
    // عرض سجل المعاملات
    fun printTransactionHistory() {
        println("=== سجل معاملات الحساب رقم $accountNumber ===")
        if (transactions.isEmpty()) {
            println("لا توجد معاملات")
        } else {
            transactions.forEach { println(it) }
        }
        println("الرصيد الحالي: $balance")
        println("==========================================")
    }
    
    // عرض معلومات الحساب
    fun displayInfo() {
        println("رقم الحساب: $accountNumber")
        println("اسم المالك: $ownerName")
        println("الرصيد: $balance")
    }
}

// كلاس البنك لإدارة الحسابات
class Bank(val name: String) {
    private val accounts = mutableMapOf<String, BankAccount>()
    
    // إنشاء حساب جديد
    fun createAccount(accountNumber: String, ownerName: String, initialBalance: Double = 0.0): BankAccount {
        if (accounts.containsKey(accountNumber)) {
            throw IllegalArgumentException("الحساب برقم $accountNumber موجود بالفعل")
        }
        
        val newAccount = BankAccount(accountNumber, ownerName, initialBalance)
        accounts[accountNumber] = newAccount
        println("تم إنشاء حساب جديد لـ $ownerName برقم $accountNumber")
        return newAccount
    }
    
    // الحصول على حساب بواسطة رقم الحساب
    fun getAccount(accountNumber: String): BankAccount? {
        return accounts[accountNumber]
    }
    
    // عرض جميع الحسابات
    fun displayAllAccounts() {
        println("=== جميع الحسابات في بنك $name ===")
        if (accounts.isEmpty()) {
            println("لا توجد حسابات حاليًا")
        } else {
            accounts.values.forEach { account ->
                account.displayInfo()
                println("------------------")
            }
        }
    }
    
    // إغلاق حساب
    fun closeAccount(accountNumber: String): Boolean {
        val account = accounts[accountNumber]
        return if (account != null) {
            accounts.remove(accountNumber)
            println("تم إغلاق حساب ${account.ownerName} برقم $accountNumber")
            true
        } else {
            println("لا يوجد حساب برقم $accountNumber")
            false
        }
    }
    
    // حساب إجمالي الودائع في البنك
    fun calculateTotalDeposits(): Double {
        return accounts.values.sumOf { it.getBalance() }
    }
}

// اختبار النظام البنكي
fun main() {
    // إنشاء بنك
    val bank = Bank("البنك المركزي")
    
    // إنشاء حسابات
    val account1 = bank.createAccount("AC001", "عمر محمد", 1000.0)
    val account2 = bank.createAccount("AC002", "سمية أحمد", 500.0)
    val account3 = bank.createAccount("AC003", "خالد إبراهيم")
    
    // عرض جميع الحسابات
    bank.displayAllAccounts()
    
    // إجراء بعض المعاملات
    account1.deposit(500.0)
    account1.withdraw(200.0)
    account2.deposit(300.0)
    account1.transfer(account2, 100.0)
    account3.deposit(1000.0)
    
    // عرض سجل المعاملات
    account1.printTransactionHistory()
    account2.printTransactionHistory()
    
    // حساب إجمالي الودائع
    val totalDeposits = bank.calculateTotalDeposits()
    println("إجمالي الودائع في البنك: $totalDeposits")
    
    // إغلاق حساب
    bank.closeAccount("AC003")
    
    // عرض الحسابات المتبقية
    bank.displayAllAccounts()
}

شرح المشروع:

هذا المشروع يمثل محاكاة لنظام بنكي بسيط باستخدام مفاهيم الكائنات والكلاسات في كوتلن:

  • BankAccount: يمثل حسابًا بنكيًا فرديًا مع وظائف مثل الإيداع والسحب والتحويل بين الحسابات.
  • Bank: يدير مجموعة من الحسابات ويوفر وظائف مثل إنشاء حساب جديد وإغلاق حساب وعرض جميع الحسابات.

المشروع يوضح مفاهيم مهمة مثل:

  • دوال البناء (Constructors) والبلوك الأولي (init): تنفيذ الشيفرة عند إنشاء الكائن.
  • إخفاء البيانات (Data Hiding): خصائص خاصة (private) مثل الرصيد وسجل المعاملات.
  • التفاعل بين الكائنات: مثل تحويل الأموال بين الحسابات.
  • الاستثناءات والتحقق من الأخطاء: التحقق من صحة المعاملات مثل التأكد من وجود رصيد كافٍ للسحب.
