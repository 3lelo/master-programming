متغيرات جافا

المتغيرات في جافا هي حاويات تُستخدم لتخزين قيم البيانات. لكل متغير نوع يحدد نوع البيانات التي يمكنه تخزينها:

  • الأنواع الأولية: تشمل int، double، boolean والمزيد.
  • الأنواع المرجعية: تشمل الكائنات والمصفوفات.
// مثال على المتغيرات
int age = 25;
double salary = 55000.50;
boolean isJavaFun = true;

تحويل الأنواع في جافا

يُستخدم تحويل الأنواع في جافا لتحويل المتغيرات من نوع إلى آخر. هناك نوعان:

  • التحويل الضمني: تحويل نوعي تلقائي (مثل int إلى double).
  • التحويل الصريح: تحويل نوعي يدوي باستخدام الأقواس (مثل double إلى int).
// مثال على تحويل الأنواع
int num = 10;
double numDouble = num; // تحويل ضمني
int numInt = (int) numDouble; // تحويل صريح

العوامل في جافا

العوامل في جافا هي رموز خاصة تُستخدم لإجراء العمليات على المتغيرات والقيم. وهي أساسية لتنفيذ الحسابات والمقارنات والمعالجات المنطقية.

أنواع العوامل

  • عوامل حسابية: تُستخدم لإجراء العمليات الرياضية 
    • int a = 10, b = 5;
int sum = a + b;     // الجمع: 15
int diff = a - b;    // الطرح: 5
int prod = a * b;    // الضرب: 50
int div = a / b;     // القسمة: 2
int mod = a % b;     // الباقي: 0
  • عوامل مقارنة: تُستخدم لمقارنة قيمتين 
    • int x = 5, y = 10;
boolean isEqual = (x == y);    // يساوي: false
boolean isNotEqual = (x != y); // لا يساوي: true
boolean isGreater = (x > y);   // أكبر من: false
boolean isLess = (x < y);      // أصغر من: true
boolean isGreaterOrEqual = (x >= y); // أكبر أو يساوي: false
boolean isLessOrEqual = (x <= y);    // أصغر أو يساوي: true
  • عوامل منطقية: تُستخدم للجمع بين العبارات الشرطية 
    • boolean a = true, b = false;
boolean andResult = a && b;    // و منطقي: false
boolean orResult = a || b;     // أو منطقي: true
boolean notResult = !a;        // نفي منطقي: false
  • عوامل بتية: تُستخدم لإجراء العمليات على مستوى البت 
    • int num1 = 5;  // ثنائي: 0101
int num2 = 3;  // ثنائي: 0011
int andBit = num1 & num2;   // و بت: 1 (0001)
int orBit = num1 | num2;    // أو بت: 7 (0111)
int xorBit = num1 ^ num2;   // إكس أو بت: 6 (0110)
  • عوامل الإسناد: تُستخدم لإسناد القيم إلى المتغيرات 
    • int a = 10;
a += 5;  // يعادل: a = a + 5 (الآن a يساوي 15)
a -= 3;  // يعادل: a = a - 3 (الآن a يساوي 12)
a *= 2;  // يعادل: a = a * 2 (الآن a يساوي 24)
a /= 4;  // يعادل: a = a / 4 (الآن a يساوي 6)

String sub = greeting.substring(0, 5); // استرجاع جزء من السلسلة من الحرف 0 إلى 5

// استبدال الأحرف
String replaced = greeting.replace('l', 'x'); // استبدال الأحرف 'l' بـ 'x' في السلسلة

عبارة Switch في جافا

توفر عبارة Switch بديلاً لعدة عبارات If-Else لمقارنة متغير مقابل قيم متعددة.

// عبارة Switch الأساسية
int day = 3;

        System.out.println("Wednesday");

    case 4:
        System.out.println("Thursday");
        break;
    case 5:
        System.out.println("Friday");
        break;
    default:
        System.out.println("Weekend");
}

  • الحلقات For-Each رائعة للتكرار البسيط لكنها لا تستطيع تعديل المجموعة

Java كسر واستمرار

الكسر والاستمرار هما عبارات تحكم تساعد في إدارة تنفيذ الحلقات وتوفير مرونة أكبر في عمليات التكرار.

عبارة الكسر

// الخروج من الحلقة
for (int i = 0; i < 10; i++) {
    if (i == 5) {
        break; // يخرج من الحلقة بالكامل عندما تكون i تساوي 5
    }
    System.out.println("الرقم الحالي: " + i);
}
// المخرجات: 0، 1، 2، 3، 4

// الكسر في الحلقات المتداخلة
for (int i = 0; i < 3; i++) {
    for (int j = 0; j < 3; j++) {
        if (i == 1 && j == 1) {
            break; // يكسر الحلقة الداخلية فقط
        }
        System.out.println("i: " + i + ", j: " + j);
    }
}

// الكسر باستخدام تصنيف محدد
outer: for (int i = 0; i < 3; i++) {
    for (int j = 0; j < 3; j++) {
        if (i == 1 && j == 1) {
            break outer; // يكسر الحلقة الخارجية
        }
        System.out.println("i: " + i + ", j: " + j);
    }
}

عبارة الاستمرار

// تخطي التكرار
for (int i = 0; i < 10; i++) {
    if (i % 2 == 0) {
        continue; // يتخطى الأرقام الزوجية
    }
    System.out.println("رقم فردي: " + i);
}

الدوال في جافا

تُستخدم الدوال (Methods) في جافا لتنظيم الكود وجعله أكثر قابلية لإعادة الاستخدام. يتم تعريف الدالة مرة واحدة ثم استدعاؤها عدة مرات عند الحاجة.

إنشاء دالة

// إنشاء دالة تقوم بطباعة رسالة
public void sayHello() {
    System.out.println("Hello, Java!");
}

// استدعاء الدالة
public static void main(String[] args) {
    sayHello();
}

دالة تُرجع قيمة

// دالة تقوم بجمع رقمين وتُرجع النتيجة
public int add(int num1, int num2) {
    return num1 + num2;
}

// استدعاء الدالة
public static void main(String[] args) {
    int result = add(5, 3);
    System.out.println("Result: " + result);
}

دالة لا تُرجع قيمة (void)

// دالة تقوم بطباعة رسالة بدون إرجاع قيمة
public void printMessage() {
    System.out.println("This is a void method!");
}

// استدعاء الدالة
public static void main(String[] args) {
    printMessage();
}

تمرير المعاملات إلى الدالة

// دالة تقوم بحساب مساحة المستطيل
public int calculateArea(int length, int width) {
    return length * width;
}

// استدعاء الدالة
public static void main(String[] args) {
    int area = calculateArea(5, 10);
    System.out.println("Area: " + area);
}

دوال مُثقلة (Method Overloading)

// تعريف دالتين بنفس الاسم ولكن بمعاملات مختلفة
public int multiply(int a, int b) {
    return a * b;
}

public double multiply(double a, double b) {
    return a * b;
}

// استدعاء الدوال
public static void main(String[] args) {
    System.out.println("Multiplication (int): " + multiply(4, 5));
    System.out.println("Multiplication (double): " + multiply(4.5, 3.2));
}

الدوال الثابتة (Static Methods)

// دالة ثابتة يمكن استدعاؤها بدون إنشاء كائن من الكلاس
public static void staticMethod() {
    System.out.println("This is a static method!");
}

// استدعاء الدالة
public static void main(String[] args) {
    staticMethod();
}

نصيحة احترافية: يُفضل استخدام الدوال لتقسيم الكود إلى وحدات صغيرة لجعله أكثر تنظيمًا وسهولة في الصيانة.

الفئات والكائنات في جافا (Classes and Objects)

الفئات والكائنات هي الأساس في البرمجة كائنية التوجه. الفئة هي المخطط الذي يحدد هيكل وسلوك الكائنات.

تعريف الفئة

public class Car {
    // متغيرات النسخة (Instance Variables)
    private String brand;
    private String model;
    private int year;

    // دالة (Method)
    public void startEngine() {
        System.out.println("The " + brand + " " + model + " is starting...");
    }

    // دالة أخرى
    public void drive() {
        System.out.println("Driving the " + year + " " + brand + " " + model);
    }

    // دوال الضبط والحصول (Setters and Getters)
    public void setBrand(String brand) {
        this.brand = brand;
    }

    public String getBrand() {
        return brand;
    }
}

إنشاء واستخدام الكائنات

public class Main {
    public static void main(String[] args) {
        // إنشاء كائن جديد
        Car myCar = new Car();
        
        // تعيين قيم للمتغيرات
        myCar.setBrand("Toyota");
        myCar.setModel("Camry");
        myCar.setYear(2024);
        
        // استدعاء الدوال
        myCar.startEngine();  // Output: The Toyota Camry is starting...
        myCar.drive();       // Output: Driving the 2024 Toyota Camry
        
        // إنشاء كائن آخر
        Car anotherCar = new Car();
        anotherCar.setBrand("Honda");
        anotherCar.setModel("Civic");
        anotherCar.setYear(2023);
    }
}

مفاهيم مهمة:
  • متغيرات النسخة (Instance Variables): تخزن حالة الكائن
  • الدوال (Methods): تحدد سلوك الكائن
  • التغليف (Encapsulation): استخدام private مع getters/setters
  • الكائن (Object): نسخة من الفئة تحتوي على قيمها الخاصة

public class BankAccount {
    private String accountNumber;
    private double balance;
    private String ownerName;

    public void setAccountNumber(String accountNumber) {

            this.accountNumber = accountNumber;
        } else {
            throw new IllegalArgumentException("Account number must be 10 digits");
        }
    }
    
    // التحقق من صحة المبلغ
    public void deposit(double amount) {
        if (amount > 0) {
            this.balance += amount;
            System.out.println("Deposited: $" + amount);
        } else {
            throw new IllegalArgumentException("Amount must be positive");
        }
    }
    
    // دالة السحب مع التحقق من الرصيد
    public void withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            this.balance -= amount;
            System.out.println("Withdrawn: $" + amount);
        } else {
            throw new IllegalArgumentException("Invalid withdrawal amount");
        }
    }
    
    // دالة عرض معلومات الحساب
    public void displayInfo() {
        System.out.println("Account Information:");
        System.out.println("Owner: " + ownerName);
        System.out.println("Account Number: " + accountNumber);
        System.out.println("Balance: $" + balance);
    }
}

  • اجعل متغيرات النسخة private دائماً للحفاظ على التغليف
  • قم بتنفيذ التحقق من صحة البيانات في setters
  • استخدم أسماء وصفية للفئات والمتغيرات والدوال
  • قم بتوثيق الكود باستخدام التعليقات عند الحاجة

الدوال في جافا (Methods)

الدوال هي كتل من التعليمات البرمجية التي تؤدي مهمة محددة. تُستخدم لتنظيم الكود وإعادة استخدامه.

أنواع الدوال

public class Calculator {
    // دالة تُرجع قيمة
    public int add(int a, int b) {
        return a + b;
    }
    
    // دالة لا تُرجع قيمة (void)
    public void printSum(int a, int b) {
        System.out.println("Sum: " + (a + b));
    }
    
    // دالة ثابتة (static)
    public static double calculateAverage(double... numbers) {
        if (numbers.length == 0) return 0.0;
        
        double sum = 0;
        for (double num : numbers) {
            sum += num;
        }
        return sum / numbers.length;
    }
    
    // دالة تأخذ مصفوفة كمعامل
    public int findMax(int[] numbers) {
        if (numbers == null || numbers.length == 0) {
            throw new IllegalArgumentException("Array cannot be empty");
        }
        
        int max = numbers[0];
        for (int num : numbers) {
            if (num > max) {
                max = num;
            }
        }
        return max;
    }
}

مثال متقدم: دوال مع معالجة الأخطاء

public class BankAccount {
    private double balance;
    
    // دالة تتحقق من صحة المعاملات
    public void deposit(double amount) throws IllegalArgumentException {
        if (amount <= 0) {
            throw new IllegalArgumentException("Amount must be positive");
        }
        this.balance += amount;
    }
    
    // دالة مع منطق معالجة متقدم
    public boolean transfer(BankAccount recipient, double amount) {
        try {
            if (amount <= 0) {
                throw new IllegalArgumentException("Transfer amount must be positive");
            }
            
            if (this.balance < amount) {
                throw new InsufficientFundsException("Insufficient funds for transfer");
            }
            
            this.balance -= amount;
            recipient.deposit(amount);
            return true;
            
        } catch (Exception e) {
            System.err.println("Transfer failed: " + e.getMessage());
            return false;
        }
    }
    
    // دالة مساعدة (helper method) خاصة
    private void validateAmount(double amount) {
        if (amount <= 0) {
            throw new IllegalArgumentException("Amount must be positive");
        }
    }
}

نقاط مهمة عن الدوال:
  • يجب أن يكون للدالة اسم وصفي يعبر عن وظيفتها
  • يمكن تحميل الدوال بتغيير عدد أو نوع المعاملات
  • الدوال الثابتة (static) تنتمي للفئة وليس للكائن
  • الدوال الخاصة (private) تستخدم للمساعدة في التنظيم الداخلي

معدلات الوصول في جافا (Access Modifiers)

معدلات الوصول تتحكم في إمكانية الوصول إلى الفئات والمتغيرات والدوال من أجزاء مختلفة من البرنامج.

أنواع معدلات الوصول

package com.example;

    private String privateVar;    // يمكن الوصول فقط داخل الفئة

    }

مثال على التغليف (Encapsulation)

public class Person {
    private String name;
    private int age;
    private double salary;
    
    // دوال الوصول العامة (Getters)
    public String getName() {
        return name;
    }
    
    public int getAge() {
        return age;
    }
    
    // دوال التعيين العامة (Setters) مع التحقق
    public void setName(String name) {
        if (name != null && !name.trim().isEmpty()) {
            this.name = name;
        } else {
            throw new IllegalArgumentException("Name cannot be empty");
        }
    }
    
    public void setAge(int age) {
        if (age >= 0 && age <= 120) {
            this.age = age;
        } else {
            throw new IllegalArgumentException("Invalid age");
        }
    }
    
    // دالة خاصة للتحقق من البيانات
    private boolean validateSalary(double salary) {
        return salary >= 0;
    }
    
    // دالة محمية يمكن للفئات الوارثة استخدامها
    protected void updateSalary(double newSalary) {
        if (validateSalary(newSalary)) {
            this.salary = newSalary;
        } else {
            throw new IllegalArgumentException("Invalid salary amount");
        }
    }
}

  • public: يمكن الوصول من أي مكان
  • private: يمكن الوصول فقط داخل نفس الفئة
  • protected: يمكن الوصول في نفس الحزمة والفئات الوارثة
  • default: يمكن الوصول في نفس الحزمة فقط

مثال عملي: تطبيق معدلات الوصول

package com.example.banking;

public class BankAccount {
    private String accountNumber;
    private double balance;
    protected String accountType;
    String branchCode;  // default access
    
    // Constructor
    public BankAccount(String accountNumber, String accountType) {
        this.accountNumber = accountNumber;
        this.accountType = accountType;
        this.balance = 0.0;
    }
    
    // Public interface
    public void deposit(double amount) {
        validateAmount(amount);
        updateBalance(amount);
    }
    
    public void withdraw(double amount) {
        validateAmount(amount);
        validateWithdrawal(amount);
        updateBalance(-amount);
    }
    
    // Protected method for inheritance
    protected void updateBalance(double amount) {
        this.balance += amount;
    }
    
    // Private validation methods
    private void validateAmount(double amount) {
        if (amount <= 0) {
            throw new IllegalArgumentException("Amount must be positive");
        }
    }
    
    private void validateWithdrawal(double amount) {
        if (amount > balance) {
            throw new InsufficientFundsException("Insufficient funds");
        }
    }
    
    // Default access method
    void updateBranchCode(String newCode) {
        this.branchCode = newCode;
    }
}

الوراثة في جافا (Inheritance)

الوراثة هي أحد أهم مفاهيم البرمجة كائنية التوجه، حيث تسمح للفئات بوراثة الخصائص والسلوكيات من فئات أخرى. تساعد في تنظيم الكود وإعادة استخدامه وبناء تسلسل هرمي للفئات.

أنواع الوراثة في جافا
  • الوراثة الأحادية (Single Inheritance): فئة واحدة ترث من فئة واحدة
  • الوراثة المتعددة المستويات (Multilevel Inheritance): سلسلة من الوراثة (A ← B ← C)
  • الوراثة الهرمية (Hierarchical Inheritance): عدة فئات ترث من فئة واحدة

ملاحظة هامة: جافا لا تدعم الوراثة المتعددة للفئات (Multiple Inheritance) لتجنب مشكلة الماس (Diamond Problem).

مثال تفصيلي للوراثة

// الفئة الأساسية
public class Vehicle {
    protected String brand;
    protected String model;
    protected int year;
    
    public Vehicle(String brand, String model, int year) {
        this.brand = brand;
        this.model = model;
        this.year = year;
    }
    
    public void start() {
        System.out.println("Vehicle starting...");
    }
    
    public void stop() {
        System.out.println("Vehicle stopping...");
    }
    
    // دالة نهائية لا يمكن تجاوزها في الفئات المشتقة
    public final void displayInfo() {
        System.out.println("Brand: " + brand + ", Model: " + model + ", Year: " + year);
    }
}

// فئة مشتقة
public class Car extends Vehicle {
    private int numberOfDoors;
    private String transmission;
    
    public Car(String brand, String model, int year, int numberOfDoors, String transmission) {
        super(brand, model, year);  // استدعاء constructor الفئة الأساسية
        this.numberOfDoors = numberOfDoors;
        this.transmission = transmission;
    }
    
    // تجاوز دالة من الفئة الأساسية
    @Override
    public void start() {
        System.out.println("Checking seatbelts...");
        super.start();  // استدعاء نسخة الفئة الأساسية
        System.out.println("Car is ready to drive!");
    }
    
    // إضافة دوال جديدة
    public void honk() {
        System.out.println("Beep! Beep!");
    }
}

    private int range;

                        int numberOfDoors, String transmission,

        super(brand, model, year, numberOfDoors, transmission);
        this.batteryCapacity = batteryCapacity;
        this.range = range;
    }

    public void start() {
        System.out.println("Checking battery level...");

        System.out.println("Electric systems initialized!");

أفضل الممارسات في الوراثة:
  • استخدم الكلمة المفتاحية final لمنع الوراثة عندما لا تكون ضرورية
  • تجنب سلاسل الوراثة العميقة (أكثر من 3 مستويات)
  • استخدم @Override دائماً عند تجاوز الدوال
  • لا تنسى استدعاء constructor الفئة الأساسية باستخدام super()
  • فكر في استخدام التركيب (Composition) بدلاً من الوراثة إذا لم تكن العلاقة "is-a" واضحة

الفئات المجردة في جافا (Abstract Classes)

الفئات المجردة هي فئات خاصة تستخدم كقوالب للفئات الأخرى. لا يمكن إنشاء كائنات منها مباشرة، ولكنها تستخدم كأساس للفئات المشتقة.

الفرق بين الفئات المجردة والعادية:
  • لا يمكن إنشاء كائنات من الفئات المجردة
  • يمكن أن تحتوي على دوال مجردة وعادية
  • يمكن أن تحتوي على متغيرات ودوال بناء
  • الفئات المشتقة يجب أن تنفذ جميع الدوال المجردة

مثال شامل للفئات المجردة

// فئة مجردة تمثل شكل هندسي
public abstract class Shape {
    protected String color;
    protected boolean filled;
    
    // يمكن للفئات المجردة أن تحتوي على constructor
    public Shape(String color, boolean filled) {
        this.color = color;
        this.filled = filled;
    }
    
    // دوال مجردة يجب تنفيذها في الفئات المشتقة
    public abstract double getArea();
    public abstract double getPerimeter();
    
    // دوال عادية مع تنفيذ
    public String getColor() {
        return color;
    }
    
    public void setColor(String color) {
        this.color = color;
    }
    
    public boolean isFilled() {
        return filled;
    }
    
    public void setFilled(boolean filled) {
        this.filled = filled;
    }
    
    // تجاوز دالة toString
    @Override
    public String toString() {
        return "A shape with color of " + color + " and " + 
                (filled ? "filled" : "not filled");
    }
}

// فئة مشتقة تمثل مستطيل
public class Rectangle extends Shape {
    protected double width;
    protected double length;
    
    public Rectangle(String color, boolean filled, double width, double length) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    
    @Override
    public double getArea() {
        return width * length;
    }
    
    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }
    
    @Override
    public String toString() {
        return "A Rectangle with width=" + width + " and length=" + length + 
                ", which is a subclass of " + super.toString();
    }
}

// فئة مشتقة تمثل دائرة
public class Circle extends Shape {
    protected double radius;
    
    public Circle(String color, boolean filled, double radius) {
        super(color, filled);
        this.radius = radius;
    }
    
    @Override
    public double getArea() {
        return Math.PI * radius * radius;
    }
    
    @Override
    public double getPerimeter() {
        return 2 * Math.PI * radius;
    }
    
    @Override
    public String toString() {
        return "A Circle with radius=" + radius + 
                ", which is a subclass of " + super.toString();
    }
}

الواجهات في جافا (Interfaces)

الواجهات هي عقد يحدد مجموعة من السلوكيات التي يجب على الفئة تنفيذها. تعتبر الواجهات أساسية في تحقيق التجريد وتعدد الوراثة في جافا.

مميزات الواجهات في جافا:
  • تحقيق تعدد الوراثة للسلوك
  • تحقيق التجريد الكامل
  • تعريف عقد موحد للفئات غير المرتبطة
  • دعم للدوال الافتراضية (منذ Java 8)
  • دعم للدوال الثابتة (static)

مثال شامل للواجهات

// واجهة أساسية للدفع
public interface Payable {
    double calculatePayment();  // دالة مجردة
    
    // دالة افتراضية (منذ Java 8)
    default void processPayment() {
        double amount = calculatePayment();
        System.out.println("Processing payment of: $" + amount);
    }
    
    // دالة ثابتة
    static boolean validateAmount(double amount) {
        return amount > 0;
    }
}

// واجهة للكائنات القابلة للمقارنة
public interface Comparable {
    int compareTo(T other);
}

// واجهة للتقارير
public interface Reportable {
    void generateReport();
    String getReportFormat();
}

// فئة تنفذ عدة واجهات
public class Employee implements Payable, Comparable, Reportable {
    private String name;
    private double salary;
    private String department;
    
    public Employee(String name, double salary, String department) {
        this.name = name;
        this.salary = salary;
        this.department = department;
    }
    
    @Override
    public double calculatePayment() {
        return salary;
    }
    
    @Override
    public int compareTo(Employee other) {
        return Double.compare(this.salary, other.salary);
    }
    
    @Override
    public void generateReport() {
        System.out.println("Employee Report:");
        System.out.println("Name: " + name);
        System.out.println("Department: " + department);
        System.out.println("Salary: $" + salary);
    }
    
    @Override
    public String getReportFormat() {
        return "PDF";
    }
}

// مثال على واجهة وظيفية
@FunctionalInterface
public interface MathOperation {
    double operate(double x, double y);
    
    // يمكن إضافة دوال افتراضية وثابتة
    default double safeDivide(double x, double y) {
        if (y == 0) return 0;
        return operate(x, y);
    }
}

        // إنشاء موظفين
        Employee emp1 = new Employee("John", 50000, "IT");
        Employee emp2 = new Employee("Alice", 60000, "HR");

  • احتفظ بالواجهات صغيرة ومركزة (مبدأ الفصل بين الواجهات)
  • استخدم الدوال الافتراضية بحذر لتجنب التعقيد
  • وثّق الواجهات جيداً لتوضيح العقد المتوقع

تعدد الأشكال في جافا (Polymorphism)

تعدد الأشكال هو القدرة على التعامل مع الكائنات من أنواع مختلفة بطريقة موحدة. يعتبر من أهم مفاهيم البرمجة كائنية التوجه.

أنواع تعدد الأشكال:
  • تعدد الأشكال في وقت التشغيل (Runtime/Dynamic Polymorphism)
  • تعدد الأشكال في وقت التجميع (Compile-time/Static Polymorphism)

مثال شامل لتعدد الأشكال

// الفئة الأساسية
public class Animal {
    protected String name;
    protected int age;
    
    public Animal(String name, int age) {
        this.name = name;
        this.age = age;
    }
    
    public void makeSound() {
        System.out.println("Some generic sound");
    }
    
    public void move() {
        System.out.println("Moving...");
    }
}

// فئات مشتقة
public class Cat extends Animal {
    private boolean isIndoor;
    
    public Cat(String name, int age, boolean isIndoor) {
        super(name, age);
        this.isIndoor = isIndoor;
    }
    
    @Override
    public void makeSound() {
        System.out.println("Meow!");
    }
    
    @Override
    public void move() {
        System.out.println("Cat is walking gracefully");
    }
    
    // تحميل الدوال (overloading)
    public void makeSound(boolean isHungry) {
        if (isHungry) {
            System.out.println("Loud Meow!!!!");
        } else {
            makeSound();
        }
    }
}

public class Bird extends Animal {
    private double wingspan;
    
    public Bird(String name, int age, double wingspan) {
        super(name, age);
        this.wingspan = wingspan;
    }
    
    @Override
    public void makeSound() {
        System.out.println("Tweet tweet!");
    }
    
    @Override
    public void move() {
        System.out.println("Bird is flying");
    }
    
    // دالة خاصة بالطيور
    public void glide() {
        System.out.println("Bird is gliding with wingspan: " + wingspan);
    }
}

// مدير للحيوانات يوضح تعدد الأشكال
public class AnimalManager {
    // تعدد الأشكال في المعاملات
    public void makeAnimalSound(Animal animal) {
        System.out.println("Processing " + animal.getClass().getSimpleName());
        animal.makeSound();
    }
    
    // تحميل الدوال (overloading)
    public void makeAnimalSound(Animal[] animals) {
        for (Animal animal : animals) {
            makeAnimalSound(animal);
        }
    }
    
    // استخدام تعدد الأشكال مع الواجهات
    public void processMovable(List movables) {
        for (Animal movable : movables) {
            movable.move();
        }
    }
}

مثال على استخدام تعدد الأشكال

public class PolymorphismDemo {
    public static void main(String[] args) {
        // إنشاء كائنات
        Animal genericAnimal = new Animal("Generic", 1);
        Animal cat = new Cat("Whiskers", 3, true);
        Animal bird = new Bird("Tweety", 2, 20.5);
        
        // تعدد الأشكال في وقت التشغيل
        System.out.println("=== Runtime Polymorphism ===");
        genericAnimal.makeSound();  // Some generic sound
        cat.makeSound();           // Meow!
        bird.makeSound();          // Tweet tweet!
        
        // استخدام مدير الحيوانات
        AnimalManager manager = new AnimalManager();
        
        // تعدد الأشكال مع المصفوفات
        System.out.println("\n=== Array Polymorphism ===");
        Animal[] animals = {genericAnimal, cat, bird};
        manager.makeAnimalSound(animals);
        
        // تعدد الأشكال مع القوائم
        System.out.println("\n=== List Polymorphism ===");
        List animalList = Arrays.asList(animals);
        manager.processMovable(animalList);
        
        // تحميل الدوال (Compile-time Polymorphism)
        System.out.println("\n=== Method Overloading ===");
        Cat realCat = (Cat) cat;  // type casting
        realCat.makeSound();      // normal sound
        realCat.makeSound(true);  // hungry sound
        
        // استخدام instanceof للتحقق من النوع
        System.out.println("\n=== Type Checking ===");
        for (Animal animal : animals) {
            if (animal instanceof Bird) {
                Bird realBird = (Bird) animal;
                realBird.glide();
            }
        }
    }
}

  • يقلل من التكرار في الكود
  • يزيد من قابلية إعادة الاستخدام
  • يحسن المرونة والصيانة
  • يدعم مبدأ التجريد
  • يسهل اختبار وتوسيع التطبيق
